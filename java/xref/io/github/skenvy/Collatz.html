136   /**
137    * Handles the sanity check for the parameterisation (P,a,b) required by both
138    * the function and reverse function.
139    *
140    * @param p Modulus used to devide n, iff n is equivalent to (0 mod P)
141    * @param a Factor by which to multiply n.
142    * @param b Value to add to the scaled value of n.
143    * @throws FailedSaneParameterCheck if P or a are 0.
144    */
145   private static void assertSaneParameterisation(BigInteger p, BigInteger a, BigInteger b) throws FailedSaneParameterCheck {
146     /* Sanity check (P,a,b) ~ P absolutely can't be 0. a "could" be zero
147      * theoretically, although would violate the reversability (if ~a is 0 then a
148      * value of "b" as the input to the reverse function would have a pre-emptive
149      * value of every number not divisible by P). The function doesn't _have_ to
150      * be reversable, but we are only interested in dealing with the class of
151      * functions that exhibit behaviour consistant with the collatz function. If
152      * _every_ input not divisable by P went straight to "b", it would simply
153      * cause a cycle consisting of "b" and every b/P^z that is an integer. While
154      * P in [-1, 1] could also be a reasonable check, as it makes every value
155      * either a 1 or 2 length cycle, it's not strictly an illegal operation.
156      * "b" being zero would cause behaviour not consistant with the collatz
157      * function, but would not violate the reversability, so no check either. */
158     if (p == BigInteger.ZERO) {
159       throw new FailedSaneParameterCheck(SaneParameterErrMsg.SANE_PARAMS_P);
160     } else if (a == BigInteger.ZERO) {
161       throw new FailedSaneParameterCheck(SaneParameterErrMsg.SANE_PARAMS_A);
162     }
163   }

165   /**
166    * Returns the output of a single application of a Collatz-esque function.
167    *
168    * @param n The value on which to perform the Collatz-esque function.
169    * @param p Modulus used to devide n, iff n is equivalent to (0 mod P).
170    * @param a Factor by which to multiply n.
171    * @param b Value to add to the scaled value of n.
172    * @return The result of the function
173    * @throws FailedSaneParameterCheck if P or a are 0.
174    */
175   public static BigInteger function(BigInteger n, BigInteger p, BigInteger a, BigInteger b) throws FailedSaneParameterCheck {
176     assertSaneParameterisation(p, a, b);
177     if (n.remainder(p) == BigInteger.ZERO) {
178       return n.divide(p);
179     } else {
180       return n.multiply(a).add(b);
181     }
182   }

195   /**
196    * Returns the output of a single application of a Collatz-esque reverse function. If
197    * only one value is returned, it is the value that would be divided by P. If two values
198    * are returned, the first is the value that would be divided by P, and the second value
199    * is that which would undergo the multiply and add step, regardless of which is larger.
200    *
201    * @param n The value on which to perform the reverse Collatz function.
202    * @param p Modulus used to devide n, iff n is equivalent to (0 mod P).
203    * @param a Factor by which to multiply n.
204    * @param b Value to add to the scaled value of n.
205    * @return The result of the function's reverse
206    * @throws FailedSaneParameterCheck if P or a are 0.
207    */
208   public static BigInteger[] reverseFunction(BigInteger n, BigInteger p, BigInteger a, BigInteger b) throws FailedSaneParameterCheck {
209     assertSaneParameterisation(p, a, b);
210     /*(n-b)%a == 0 && (n-b)%(P*a) != 0*/
211     if (n.subtract(b).remainder(a) == BigInteger.ZERO && n.subtract(b).remainder(p.multiply(a)) != BigInteger.ZERO) {
212       // [P*n] + [(n-b)//a]
213       BigInteger[] preVals = new BigInteger[]{p.multiply(n), n.subtract(b).divide(a)};
214       return preVals;
215     } else { // [P*n]
216       return new BigInteger[]{p.multiply(n)};
217     }
218   }

234   /**
235    * Provides the appropriate lambda to use to check if iterations on an initial
236    * value have reached either the stopping time, or total stopping time.
237    *
238    * @param n The initial value to confirm against a stopping time check.
239    * @param totalStop If false, the lambda will confirm that iterations of n
240    *     have reached the oriented stopping time to reach a value closer to 0.
241    *     If true, the lambda will simply check equality to 1.
242    * @return The lambda to check for the stopping time.
243    */
244   private static Function<BigInteger, Boolean> stoppingTimeTerminus(BigInteger n, boolean totalStop) {
245     if (totalStop) {
246       return (x) -> {
247         return x.equals(BigInteger.ONE);
248       };
249     } else if (n.compareTo(BigInteger.ZERO) >= 0) {
250       return (x) -> {
251         return ((x.compareTo(n) == -1) && (x.compareTo(BigInteger.ZERO) == 1));
252       };
253     } else {
254       return (x) -> {
255         return ((x.compareTo(n) == 1) && (x.compareTo(BigInteger.ZERO) == -1));
256       };
257     }
258   }

260   /** Contains the results of computing a hailstone sequence via {@code Collatz.hailstoneSequence(~)}. */
261   public static final class HailstoneSequence {
262 
263     /** The set of values that comprise the hailstone sequence. */
264     final BigInteger[] values;
265 
266     private final Function<BigInteger, Boolean> terminate;
267 
268     /** A terminal condition that reflects the final state of the hailstone sequencing,
269      *  whether than be that it succeeded at determining the stopping time, the total
270      *  stopping time, found a cycle, or got stuck on zero (or surpassed the max total). */
271     final SequenceState terminalCondition;
272 
273     /** A status value that has different meanings depending on what the terminal condition
274      *  was. If the sequence completed either via reaching the stopping or total stopping time,
275      *  or getting stuck on zero, then this value is the stopping/terminal time. If the sequence
276      *  got stuck on a cycle, then this value is the cycle length. If the sequencing passes the
277      *  maximum stopping time then this is the value that was provided as that maximum. */
278     final int terminalStatus;
279 
280     /**
281      * Initialise and compute a new Hailstone Sequence.
282    *
283      * @param initialValue The value to begin the hailstone sequence from.
284      * @param p Modulus used to devide n, iff n is equivalent to (0 mod P).
285      * @param a Factor by which to multiply n.
286      * @param b Value to add to the scaled value of n.
287      * @param maxTotalStoppingTime Maximum amount of times to iterate the function, if 1 is not reached.
288      * @param totalStoppingTime Whether or not to execute until the "total" stopping time
289      *     (number of iterations to obtain 1) rather than the regular stopping time (number
290      *     of iterations to reach a value less than the initial value).
291      * @throws FailedSaneParameterCheck if P or a are 0.
292      */
293     public HailstoneSequence(BigInteger initialValue, BigInteger p, BigInteger a, BigInteger b, int maxTotalStoppingTime, boolean totalStoppingTime) throws FailedSaneParameterCheck {
294       terminate = stoppingTimeTerminus(initialValue, totalStoppingTime);
295       if (initialValue.equals(BigInteger.ZERO)) {
296         // 0 is always an immediate stop.
297         values = new BigInteger[]{BigInteger.ZERO};
298         terminalCondition = SequenceState.ZERO_STOP;
299         terminalStatus = 0;
300       } else if (initialValue.equals(BigInteger.ONE)) {
301         // 1 is always an immediate stop, with 0 stopping time.
302         values = new BigInteger[]{BigInteger.ONE};
303         terminalCondition = SequenceState.TOTAL_STOPPING_TIME;
304         terminalStatus = 0;
305       } else {
306         // Otherwise, hail!
307         int minMaxTotalStoppingTime = Math.max(maxTotalStoppingTime, 1);
308         ArrayList<BigInteger> preValues = new ArrayList<BigInteger>(minMaxTotalStoppingTime + 1);
309         preValues.add(initialValue);
310         BigInteger next;
311         for (int k = 1; k <= minMaxTotalStoppingTime; k++) {
312           next = function(preValues.get(k - 1), p, a, b);
313           // Check if the next hailstone is either the stopping time, total
314           // stopping time, the same as the initial value, or stuck at zero.
315           if (terminate.apply(next)) {
316             preValues.add(next);
317             if (next.equals(BigInteger.ONE)) {
318               terminalCondition = SequenceState.TOTAL_STOPPING_TIME;
319             } else {
320               terminalCondition = SequenceState.STOPPING_TIME;
321             }
322             terminalStatus = k;
323             values = new BigInteger[preValues.size()];
324             preValues.toArray(values);
325             return;
326           }
327           if (preValues.contains(next)) {
328             preValues.add(next);
329             int cycleInit = 1;
330             for (int j = 1; j <= k; j++) {
331               if (preValues.get(k - j).equals(next)) {
332                 cycleInit = j;
333                 break;
334               }
335             }
336             terminalCondition = SequenceState.CYCLE_LENGTH;
337             terminalStatus = cycleInit;
338             values = new BigInteger[preValues.size()];
339             preValues.toArray(values);
340             return;
341           }
342           if (next.equals(BigInteger.ZERO)) {
343             preValues.add(BigInteger.ZERO);
344             terminalCondition = SequenceState.ZERO_STOP;
345             terminalStatus = -k;
346             values = new BigInteger[preValues.size()];
347             preValues.toArray(values);
348             return;
349           }
350           preValues.add(next);
351         }
352         terminalCondition = SequenceState.MAX_STOP_OUT_OF_BOUNDS;
353         terminalStatus = minMaxTotalStoppingTime;
354         values = new BigInteger[preValues.size()];
355         preValues.toArray(values);
356       }
357     }
358   }

360   /**
361    * Returns a list of successive values obtained by iterating a Collatz-esque
362    * function, until either 1 is reached, or the total amount of iterations
363    * exceeds maxTotalStoppingTime, unless totalStoppingTime is False,
364    * which will terminate the hailstone at the "stopping time" value, i.e. the
365    * first value less than the initial value. While the sequence has the
366    * capability to determine that it has encountered a cycle, the cycle from "1"
367    * wont be attempted or reported as part of a cycle, regardless of default or
368    * custom parameterisation, as "1" is considered a "total stop".
369    *
370    * @param initialValue The value to begin the hailstone sequence from.
371    * @param p Modulus used to devide n, iff n is equivalent to (0 mod P).
372    * @param a Factor by which to multiply n.
373    * @param b Value to add to the scaled value of n.
374    * @param maxTotalStoppingTime Maximum amount of times to iterate the function, if 1 is not reached.
375    * @param totalStoppingTime Whether or not to execute until the "total" stopping time
376    *     (number of iterations to obtain 1) rather than the regular stopping time (number
377    *     of iterations to reach a value less than the initial value).
378    * @return A set of values that form the hailstone sequence.
379    * @throws FailedSaneParameterCheck if P or a are 0.
380    */
381   public static HailstoneSequence hailstoneSequence(BigInteger initialValue, BigInteger p, BigInteger a, BigInteger b, int maxTotalStoppingTime, boolean totalStoppingTime) throws FailedSaneParameterCheck {
382     // Call out the function before any magic returns to trap bad values.
383     @SuppressWarnings("unused")
384     final BigInteger _throwaway = function(initialValue, p, a, b);
385     // Return the hailstone sequence.
386     return new HailstoneSequence(initialValue, p, a, b, maxTotalStoppingTime, totalStoppingTime);
387   }

402   /**
403    * Returns the stopping time, the amount of iterations required to reach a
404    * value less than the initial value, or null if maxStoppingTime is exceeded.
405    * Alternatively, if totalStoppingTime is True, then it will instead count
406    * the amount of iterations to reach 1. If the sequence does not stop, but
407    * instead ends in a cycle, the result will be (Double.POSITIVE_INFINITY).
408    * If (P,a,b) are such that it is possible to get stuck on zero, the result
409    * will be the negative of what would otherwise be the "total stopping time"
410    * to reach 1, where 0 is considered a "total stop" that should not occur as
411    * it does form a cycle of length 1.
412    *
413    * @param initialValue The value for which to find the stopping time.
414    * @param p Modulus used to devide n, iff n is equivalent to (0 mod P).
415    * @param a Factor by which to multiply n.
416    * @param b Value to add to the scaled value of n.
417    * @param maxStoppingTime Maximum amount of times to iterate the function, if
418    *     the stopping time is not reached. IF the maxStoppingTime is reached,
419    *     the function will return null.
420    * @param totalStoppingTime (bool): Whether or not to execute until the "total" stopping
421    *     time (number of iterations to obtain 1) rather than the regular stopping
422    *     time (number of iterations to reach a value less than the initial value).
423    * @return The stopping time, or, in a special case, infinity, null or a negative.
424    * @throws FailedSaneParameterCheck if P or a are 0.
425    */
426   public static Double stoppingTime(BigInteger initialValue, BigInteger p, BigInteger a, BigInteger b, int maxStoppingTime, boolean totalStoppingTime) throws FailedSaneParameterCheck {
427     /* The information is contained in the hailstone sequence. Although the "max~time"
428      * for hailstones is named for "total stopping" time and the "max~time" for this
429      * "stopping time" function is _not_ "total", they are handled the same way, as
430      * the default for "totalStoppingTime" for hailstones is true, but for this, is
431      * false. Thus the naming difference. */
432     HailstoneSequence hail = hailstoneSequence(initialValue, p, a, b, maxStoppingTime, totalStoppingTime);
433     // For total/regular/zero stopping time, the value is already the same as
434     // that present, for cycles we report infinity instead of the cycle length,
435     // and for max stop out of bounds, we report null instead of the max stop cap
436     switch (hail.terminalCondition) {
437       case TOTAL_STOPPING_TIME:
438         return (double) hail.terminalStatus;
439       case STOPPING_TIME:
440         return (double) hail.terminalStatus;
441       case CYCLE_LENGTH:
442         return Double.POSITIVE_INFINITY;
443       case ZERO_STOP:
444         return (double) hail.terminalStatus;
445       case MAX_STOP_OUT_OF_BOUNDS:
446         return null;
447       default:
448         return null;
449     }
450   }

467   /**
468    * Nodes that form a "tree graph", structured as a tree, with their own node's value,
469    * as well as references to either possible child node, where a node can only ever have
470    * two children, as there are only ever two reverse values. Also records any possible
471    * "terminal sequence state", whether that be that the "orbit distance" has been reached,
472    * as an "out of bounds" stop, which is the regularly expected terminal state. Other
473    * terminal states possible however include the cycle state and cycle length (end) states.
474    */
475   public static final class TreeGraphNode {
476 
477     /** The value of this node in the tree. */
478     final BigInteger nodeValue;
479 
480     /** The terminal state; null if not a terminal node, MAX_STOP_OUT_OF_BOUNDS if the maxOrbitDistance
481      *  has been reached, CYCLE_LENGTH if the node's value is found to have occured previously, or
482      *  CYCLE_INIT, retroactively applied when a CYCLE_LENGTH state node is found. */
483     // The only variable of TreeGraphNode to not be "final" as it must be possible to update retroactively.
484     SequenceState terminalSequenceState = null;
485 
486     /** The "Pre N/P" child of this node that is always
487      *  present if this is not a terminal node. */
488     final TreeGraphNode preNDivPNode;
489 
490     /** The "Pre aN+b" child of this node that is present
491      *  if it exists and this is not a terminal node. */
492     final TreeGraphNode preANplusBNode;
493 
494     /** A map of previous graph nodes which maps instances of
495      *  TreeGraphNode to themselves, to enable cycle detection. */
496     private final Map<BigInteger, TreeGraphNode> cycleCheck;
497 
498     /**
499      * Create an instance of TreeGraphNode which will yield its entire sub-tree of all child nodes.
500      *
501      * @param nodeValue The value for which to find the tree graph node reversal.
502      * @param maxOrbitDistance The maximum distance/orbit/branch length to travel.
503      * @param p Modulus used to devide n, iff n is equivalent to (0 mod P).
504      * @param a Factor by which to multiply n.
505      * @param b Value to add to the scaled value of n.
506      * @throws FailedSaneParameterCheck if P or a are 0.
507      */
508     public TreeGraphNode(BigInteger nodeValue, int maxOrbitDistance, BigInteger p, BigInteger a, BigInteger b) throws FailedSaneParameterCheck {
509       this.nodeValue = nodeValue;
510       if (Math.max(0, maxOrbitDistance) == 0) {
511         this.terminalSequenceState = SequenceState.MAX_STOP_OUT_OF_BOUNDS;
512         this.preNDivPNode = null;
513         this.preANplusBNode = null;
514         this.cycleCheck = null;
515       } else {
516         BigInteger[] reverses = reverseFunction(nodeValue, p, a, b);
517         cycleCheck = new HashMap<BigInteger, TreeGraphNode>();
518         this.cycleCheck.put(this.nodeValue, this);
519         this.preNDivPNode = new TreeGraphNode(reverses[0], maxOrbitDistance - 1, p, a, b, this.cycleCheck);
520         if (reverses.length == 2) {
521           this.preANplusBNode = new TreeGraphNode(reverses[1], maxOrbitDistance - 1, p, a, b, this.cycleCheck);
522         } else {
523           this.preANplusBNode = null;
524         }
525       }
526 
527     }

529     /**
530      * Create an instance of TreeGraphNode which will yield its entire sub-tree of all child nodes.
531      * This is used internally by itself and the public constructor to pass the cycle checking map,
532      * recursively determining subsequent child nodes.
533      *
534      * @param nodeValue The value for which to find the tree graph node reversal.
535      * @param maxOrbitDistance The maximum distance/orbit/branch length to travel.
536      * @param p Modulus used to devide n, iff n is equivalent to (0 mod P).
537      * @param a Factor by which to multiply n.
538      * @param b Value to add to the scaled value of n.
539      * @param cycleCheck Checks if this node's value already occurred.
540      * @return the tree graph node and its subsequent tree, for the given parameters.
541      * @throws FailedSaneParameterCheck if P or a are 0.
542      */
543     private TreeGraphNode(BigInteger nodeValue, int maxOrbitDistance, BigInteger p, BigInteger a, BigInteger b, Map<BigInteger, TreeGraphNode> cycleCheck) throws FailedSaneParameterCheck {
544       this.nodeValue = nodeValue;
545       this.cycleCheck = cycleCheck;
546       if (this.cycleCheck.keySet().contains(this.nodeValue)) {
547         this.cycleCheck.get(this.nodeValue).terminalSequenceState = SequenceState.CYCLE_INIT;
548         this.terminalSequenceState = SequenceState.CYCLE_LENGTH;
549         this.preNDivPNode = null;
550         this.preANplusBNode = null;
551       } else if (Math.max(0, maxOrbitDistance) == 0) {
552         this.terminalSequenceState = SequenceState.MAX_STOP_OUT_OF_BOUNDS;
553         this.preNDivPNode = null;
554         this.preANplusBNode = null;
555       } else {
556         this.cycleCheck.put(this.nodeValue, this);
557         this.terminalSequenceState = null;
558         BigInteger[] reverses = reverseFunction(nodeValue, p, a, b);
559         this.preNDivPNode = new TreeGraphNode(reverses[0], maxOrbitDistance - 1, p, a, b, this.cycleCheck);
560         if (reverses.length == 2) {
561           this.preANplusBNode = new TreeGraphNode(reverses[1], maxOrbitDistance - 1, p, a, b, this.cycleCheck);
562         } else {
563           this.preANplusBNode = null;
564         }
565       }
566     }
567 
568     /**
569      * Create an instance of TreeGraphNode by directly passing it the values required to instantiate,
570      * intended to be used in testing by manually creating trees in reverse, by passing expected child
571      * nodes to their parents until the entire expected tree is created.
572      *
573      * @param nodeValue The value expected of this node.
574      * @param terminalSequenceState The expected sequence state;
575      *     null, MAX_STOP_OUT_OF_BOUNDS, CYCLE_INIT or CYCLE_LENGTH.
576      * @param preNDivPNode The expected "Pre N/P" child node.
577      * @param preANplusBNode The expected "Pre aN+b" child node.
578      */
579     public TreeGraphNode(BigInteger nodeValue, SequenceState terminalSequenceState, TreeGraphNode preNDivPNode, TreeGraphNode preANplusBNode) {
580       this.nodeValue = nodeValue;
581       this.terminalSequenceState = terminalSequenceState;
582       this.preNDivPNode = preNDivPNode;
583       this.preANplusBNode = preANplusBNode;
584       this.cycleCheck = null;
585     }

587     /** The equality between TreeGraphNodes is determined exclusively by the
588      *  node's value, independent of the child nodes or sequence states that
589      *  would be relevant to the node's status relative to the tree. */
590     @Override
591     public boolean equals(Object obj) {
592       if (obj == null) {
593         return false;
594       }
595       if (obj.getClass() != this.getClass()) {
596         return false;
597       }
598       final TreeGraphNode tgn = (TreeGraphNode) obj;
599       return this.nodeValue.equals(tgn.nodeValue);
600     }
601 
602     /** The hashCode of a TreeGraphNode is determined by the
603      *  node's value, the child nodes and sequence state. */
604     @Override
605     public int hashCode() {
606       int hash = this.nodeValue.hashCode();
607       hash = 17 * hash + (this.terminalSequenceState != null ? this.terminalSequenceState.hashCode() : 0);
608       hash = 17 * hash + (this.preNDivPNode != null ? this.preNDivPNode.hashCode() : 0);
609       hash = 17 * hash + (this.preANplusBNode != null ? this.preANplusBNode.hashCode() : 0);
610       return hash;
611     }
612 
613     /**
614      * A much stricter equality check than the {@code equals(Object obj)} override.
615      * This will only confirm an equality if the whole subtree of both nodes, including
616      * node values, sequence states, and child nodes, checked recursively, are equal.
617      *
618      * @param tgn The TreeGraphNode with which to compare equality.
619      * @return {@code true}, if the entire sub-trees are equal.
620      */
621     public boolean subTreeEquals(TreeGraphNode tgn) {
622       if (!this.nodeValue.equals(tgn.nodeValue) || this.terminalSequenceState != tgn.terminalSequenceState) {
623         return false;
624       }
625       if (this.preNDivPNode == null && tgn.preNDivPNode != null) {
626         return false;
627       }
628       if (this.preNDivPNode != null && !this.preNDivPNode.subTreeEquals(tgn.preNDivPNode)) {
629         return false;
630       }
631       if (this.preANplusBNode == null && tgn.preANplusBNode != null) {
632         return false;
633       }
634       if (this.preANplusBNode != null && !this.preANplusBNode.subTreeEquals(tgn.preANplusBNode)) {
635         return false;
636       }
637       return true;
638     }
639   }

642    *  Contains the root node of a tree of {@code TreeGraphNode}'s.*/
643   public static final class TreeGraph {

645     /** The root node of the tree of {@code TreeGraphNode}'s. */
646     final TreeGraphNode root;
647 
648     /**
649      * Create a new TreeGraph with the root node defined by the inputs.
650      *
651      * @param nodeValue The value for which to find the tree graph node reversal.
652      * @param maxOrbitDistance The maximum distance/orbit/branch length to travel.
653      * @param p Modulus used to devide n, iff n is equivalent to (0 mod P).
654      * @param a Factor by which to multiply n.
655      * @param b Value to add to the scaled value of n.
656      * @throws FailedSaneParameterCheck if P or a are 0.
657      */
658     public TreeGraph(BigInteger nodeValue, int maxOrbitDistance, BigInteger p, BigInteger a, BigInteger b) throws FailedSaneParameterCheck {
659       this.root = new TreeGraphNode(nodeValue, maxOrbitDistance, p, a, b);
660     }
661 
662     /**
663      * Create a new TreeGraph by directly passing it the root node.
664      * Intended to be used in testing by manually creating trees.
665      *
666      * @param root The root node of the tree.
667      */
668     public TreeGraph(TreeGraphNode root) {
669       this.root = root;
670     }
671 
672     /** The hashCode of a TreeGraph is determined
673      *  by the hash of the root node. */
674     @Override
675     public int hashCode() {
676       return 29 * this.root.hashCode();
677     }
678 
679     /** The equality between {@code TreeGraph}'s is determined by the equality check on subtrees. 
680      *  A subtree check will be done on both {@code TreeGraph}'s root nodes. */
681     @Override
682     public boolean equals(Object obj) {
683       // Generic checks
684       if (obj == null) {
685         return false;
686       }
687       if (obj.getClass() != this.getClass()) {
688         return false;
689       }
690       // Check immediate fields
691       final TreeGraph tg = (TreeGraph) obj;
692       return this.root.subTreeEquals(tg.root);
693     }
694   }
